### Skeleton

```javascript
import { supabase } from '../shared/supabase-client.js';
import { showModal, closeModal } from '../shared/modal.js';
import { showToast } from '../shared/toast.js';

export default function render() {
  return `
    <div class="cohorts-page">
      <header>
        <h1>Turmas</h1>
        <button id="btn-new" class="btn-primary">+ Nova Turma</button>
      </header>
      <div class="filters">
        <select id="filter-status"><option value="active">Ativas</option><option value="archived">Arquivadas</option><option value="all">Todas</option></select>
        <input id="search" placeholder="Buscar..." />
      </div>
      <table id="cohorts-table"><tbody></tbody></table>
    </div>
  `;
}

export async function init() {
  await loadCohorts();
  document.getElementById('btn-new').onclick = () => showCohortModal();
  document.getElementById('filter-status').onchange = loadCohorts;
  document.getElementById('search').oninput = debounce(loadCohorts, 300);
}

async function loadCohorts() {
  const filter = document.getElementById('filter-status').value;
  const search = document.getElementById('search').value;
  let query = supabase.from('cohorts').select('*, student_cohorts(count)');
  if (filter === 'active') query = query.eq('active', true);
  if (filter === 'archived') query = query.eq('active', false);
  if (search) query = query.ilike('name', `%${search}%`);
  const { data, error } = await query.order('name');
  // render rows
}
```
