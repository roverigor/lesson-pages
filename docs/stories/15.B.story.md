## O que fazer

Implementar edge function inbound `ac-purchase-webhook` que recebe eventos de compra do ActiveCampaign. Strategy: minimal-sync handler conforme ADR-016 §3 — valida HMAC + INSERT em `ac_purchase_events` + retorna 200 imediato. Worker pg_cron (15.A) processa async.

## Business Value

- **Trigger automático onboarding pós-compra** (FR-7) — coração do auto-fluxo AC
- **NFR-6 atendido** (webhook < 2s p95) via minimal-sync + idempotência DB
- **Idempotência critical** (NFR-4 + EC-1) — UNIQUE ac_event_id previne duplicidade
- **Destrava 15.D** (UI mappings + lista eventos) — sem dados não há observabilidade

## Dev Notes

### Skeleton (referência ADR-016 §3 webhook handler)

```typescript
import { serve } from "https://deno.land/std@0.177.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2.39.0";
import { validateHmac } from "../_shared/ac-utils.ts";

const SUPABASE_URL = Deno.env.get("SUPABASE_URL")!;
const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!;
const AC_WEBHOOK_SECRET = Deno.env.get("AC_WEBHOOK_SECRET")!;

const sb = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY, { auth: { autoRefreshToken: false, persistSession: false } });

serve(async (req) => {
  const start = performance.now();
  const url = new URL(req.url);

  if (url.searchParams.get("warmup") === "1") return new Response("ok", { status: 200 });

  if (req.method === "OPTIONS") return corsResponse();
  if (req.method !== "POST") return jsonResponse({ error: "method_not_allowed" }, 405);

  const sig = req.headers.get("X-AC-Signature");
  const body = await req.text();

  if (!sig || !await validateHmac(body, sig, AC_WEBHOOK_SECRET)) {
    console.error("HMAC invalid", { hasSignature: !!sig });
    await alertSlack("AC webhook HMAC inválido");
    return jsonResponse({ error: "unauthorized" }, 401);
  }

  let payload: any;
  try { payload = JSON.parse(body); } catch { return jsonResponse({ error: "invalid_json" }, 400); }

  const acEventId = payload.event_id || `${payload.order_id}_${payload.contact_id}`;

  const { error } = await sb.from("ac_purchase_events").insert({ ac_event_id: acEventId, payload, status: "received" });

  if (error?.code === "23505") {
    console.log({ event: "duplicate", ac_event_id: acEventId });
    return jsonResponse({ status: "duplicate", event_id: acEventId });
  }
  if (error) {
    console.error({ event: "db_error", error: error.message });
    return jsonResponse({ error: "db_error" }, 500);
  }

  const duration = Math.round(performance.now() - start);
  console.log({ event: "accepted", ac_event_id: acEventId, duration_ms: duration });
  return jsonResponse({ status: "accepted", event_id: acEventId });
});
```

### `_shared/ac-utils.ts` (criar)

```typescript
export async function validateHmac(body: string, signature: string, secret: string): Promise<boolean> {
  const encoder = new TextEncoder();
  const key = await crypto.subtle.importKey("raw", encoder.encode(secret), { name: "HMAC", hash: "SHA-256" }, false, ["sign"]);
  const sigBuf = await crypto.subtle.sign("HMAC", key, encoder.encode(body));
  const expected = Array.from(new Uint8Array(sigBuf)).map(b => b.toString(16).padStart(2, "0")).join("");
  return signature === expected || signature === `sha256=${expected}`;
}
```
