### Realtime Badge Pendentes

```javascript
import { supabase } from './shared/supabase-client.js';

async function setupPendingBadge() {
  const updateBadge = async () => {
    const { count } = await supabase
      .from('pending_student_assignments')
      .select('id', { count: 'exact', head: true })
      .is('resolved_at', null);
    document.getElementById('pending-badge').textContent = count || 0;
    document.getElementById('pending-badge').style.display = count > 0 ? 'inline' : 'none';
  };

  await updateBadge();
  supabase.channel('pending-changes')
    .on('postgres_changes', { event: '*', schema: 'public', table: 'pending_student_assignments' }, updateBadge)
    .subscribe();
}
```
