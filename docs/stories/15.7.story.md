## Dev Notes

### View `student_dispatch_timeline` (criada em 15.A)

```sql
-- Já em 15.A AC-27
CREATE OR REPLACE VIEW student_dispatch_timeline AS
SELECT
  sl.id AS link_id,
  sl.student_id,
  s.name AS student_name,
  sv.name AS survey_name,
  sl.cohort_snapshot_name,
  sl.version_id,
  svv.version_number,
  sl.sent_at, sl.delivered_at, sl.read_at, sl.used_at,
  sl.send_status,
  sl.token,
  sl.meta_message_id,
  sr.submitted_at,
  sl.created_at
FROM survey_links sl
JOIN students s ON s.id = sl.student_id
JOIN surveys sv ON sv.id = sl.survey_id
LEFT JOIN survey_versions svv ON svv.id = sl.version_id
LEFT JOIN survey_responses sr ON sr.link_id = sl.id
ORDER BY sl.sent_at DESC NULLS LAST;
```
