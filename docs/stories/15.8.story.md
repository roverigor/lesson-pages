## Dev Notes

### Edge Function sync-meta-templates

```typescript
serve(async (req) => {
  const auth = await verifyAdminOrCs(req.headers.get('Authorization'));
  if (!auth) return jsonResponse({ error: 'unauthorized' }, 401);

  const wabaId = Deno.env.get('META_WABA_ID')!;
  const apiKey = Deno.env.get('META_API_KEY')!;

  const res = await fetch(`https://graph.facebook.com/v21.0/${wabaId}/message_templates?limit=200`, {
    headers: { Authorization: `Bearer ${apiKey}` }
  });
  const { data: templates } = await res.json();

  const upserts = templates.map(t => ({
    name: t.name,
    language: t.language,
    category: t.category,
    body_params_count: t.components?.find(c => c.type === 'BODY')?.example?.body_text?.[0]?.length || 0,
    button_count: t.components?.find(c => c.type === 'BUTTONS')?.buttons?.length || 0,
    status: t.status === 'APPROVED' ? 'active' : t.status.toLowerCase(),
    approved_at: t.created_time
  }));

  const { error } = await sb.from('meta_templates').upsert(upserts, { onConflict: 'name' });
  if (error) return jsonResponse({ error: error.message }, 500);
  return jsonResponse({ synced: upserts.length });
});
```

// Antes do loop dispatch
if (body.use_template) {
  const { data: tpl } = await client.from('meta_templates').select('status').eq('name', body.use_template).single();
  if (!tpl || tpl.status !== 'active') {
    return jsonResponse({ error: 'template_inactive_or_missing', template: body.use_template }, 400);
  }
}
