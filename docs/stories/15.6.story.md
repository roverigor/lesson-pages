## Dev Notes

### Skeleton Drag-Drop SortableJS

```javascript
import Sortable from 'sortablejs';

function initBuilder() {
  const palette = document.getElementById('palette');
  const canvas = document.getElementById('canvas');

  new Sortable(palette, {
    group: { name: 'questions', pull: 'clone', put: false },
    sort: false,
    onEnd: () => {} // palette items always clone
  });

  new Sortable(canvas, {
    group: 'questions',
    animation: 150,
    handle: '.drag-handle',
    onAdd: (evt) => {
      const type = evt.item.dataset.type;
      replaceWithQuestionCard(evt.item, type);
      updatePreview();
    },
    onUpdate: updatePreview // reorder triggers preview update
  });
}
```
