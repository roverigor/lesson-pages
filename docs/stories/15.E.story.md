### Função SQL Skeleton

```sql
CREATE OR REPLACE FUNCTION alert_slack_if_unhealthy()
RETURNS VOID AS $$
DECLARE
  v_total INT;
  v_failed INT;
  v_fail_rate NUMERIC;
  v_pending_old INT;
  v_callback_failures INT;
  v_last_worker TIMESTAMPTZ;
  v_message TEXT;
BEGIN
  -- 1. Webhook fail rate últimas 24h
  SELECT COUNT(*), COUNT(*) FILTER (WHERE status = 'failed')
  INTO v_total, v_failed
  FROM ac_purchase_events
  WHERE created_at > now() - INTERVAL '24 hours';

  v_fail_rate := CASE WHEN v_total > 0 THEN v_failed::NUMERIC / v_total * 100 ELSE 0 END;

  IF v_fail_rate > 10 AND v_total >= 5 AND NOT recently_alerted('webhook_fail_rate') THEN
    v_message := format('⚠️ AC Webhook fail rate %s%% últimas 24h (%s/%s falharam)', round(v_fail_rate, 1), v_failed, v_total);
    PERFORM send_slack_alert('webhook_fail_rate', v_message);
  END IF;

  -- 2. Pendentes > 24h
  SELECT COUNT(*) INTO v_pending_old
  FROM pending_student_assignments
  WHERE resolved_at IS NULL AND created_at < now() - INTERVAL '24 hours';

  IF v_pending_old > 0 AND NOT recently_alerted('pending_overdue') THEN
    v_message := format('⏰ %s aluno(s) pendente(s) há mais de 24h aguardando resolução CS', v_pending_old);
    PERFORM send_slack_alert('pending_overdue', v_message);
  END IF;

  -- 3. Worker travado (sem processar há > 5min apesar de eventos received)
  SELECT MAX(processed_at) INTO v_last_worker FROM ac_purchase_events WHERE status = 'processed';
  IF EXISTS (SELECT 1 FROM ac_purchase_events WHERE status = 'received') AND
     (v_last_worker IS NULL OR v_last_worker < now() - INTERVAL '5 minutes') AND
     NOT recently_alerted('worker_stuck') THEN
    v_message := '🚨 Worker pg_cron parece travado — eventos received não processados há > 5min';
    PERFORM send_slack_alert('worker_stuck', v_message);
  END IF;

  -- 4. Callback AC failures
  SELECT COUNT(*) INTO v_callback_failures FROM ac_dispatch_callbacks
  WHERE status = 'failed' AND last_attempt_at > now() - INTERVAL '1 hour';
  IF v_callback_failures > 5 AND NOT recently_alerted('callback_failures') THEN
    v_message := format('❌ %s callbacks AC falharam última hora (3 retries esgotados)', v_callback_failures);
    PERFORM send_slack_alert('callback_failures', v_message);
  END IF;
END;
$$ LANGUAGE plpgsql;

-- Throttle helper
CREATE TABLE IF NOT EXISTS alert_history (
  alert_key TEXT PRIMARY KEY,
  last_sent_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE OR REPLACE FUNCTION recently_alerted(p_key TEXT, p_within INTERVAL DEFAULT INTERVAL '1 hour')
RETURNS BOOLEAN AS $$
  SELECT EXISTS (SELECT 1 FROM alert_history WHERE alert_key = p_key AND last_sent_at > now() - p_within);
$$ LANGUAGE sql;

CREATE OR REPLACE FUNCTION send_slack_alert(p_key TEXT, p_message TEXT)
RETURNS VOID AS $$
BEGIN
  -- HTTP call via pg_net OR INSERT into queue table for edge function pickup
  PERFORM net.http_post(
    url := 'https://gpufcipkajppykmnmdeh.supabase.co/functions/v1/send-slack-alert',
    body := jsonb_build_object('message', p_message, 'key', p_key),
    headers := jsonb_build_object('Authorization', 'Bearer ' || current_setting('app.service_role_key'))
  );
  INSERT INTO alert_history (alert_key, last_sent_at) VALUES (p_key, now())
    ON CONFLICT (alert_key) DO UPDATE SET last_sent_at = now();
END;
$$ LANGUAGE plpgsql;
```

import { sendDM } from '../_shared/slack.ts';

});
