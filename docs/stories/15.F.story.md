### Heurística Fuzzy Match

```javascript
function fuzzyMatchCohort(productName, cohorts) {
  const norm = s => s.toLowerCase().replace(/[^a-z0-9]/g, '');
  const target = norm(productName);

  let best = null;
  let bestScore = 0;

  for (const c of cohorts) {
    const cName = norm(c.name);
    // Simple trigram overlap
    const trigrams = new Set();
    for (let i = 0; i <= target.length - 3; i++) trigrams.add(target.slice(i, i + 3));
    let matches = 0;
    for (let i = 0; i <= cName.length - 3; i++) {
      if (trigrams.has(cName.slice(i, i + 3))) matches++;
    }
    const score = matches / Math.max(trigrams.size, 1);
    if (score > bestScore) { best = c; bestScore = score; }
  }
  return bestScore > 0.7 ? best : null;
}
```

### Resolução Action

```javascript
async function resolvePending(pendingId, cohortId, surveyId, templateName, createMapping) {
  // 1. Insert student_cohorts
  await supabase.from('student_cohorts').upsert({
    student_id: pending.student_id,
    cohort_id: cohortId,
    active: true
  });

  // 2. Trigger dispatch
  await fetch(`${SUPABASE_URL}/functions/v1/dispatch-survey`, {
    method: 'POST',
    headers: { Authorization: `Bearer ${session.access_token}`, 'Content-Type': 'application/json' },
    body: JSON.stringify({ survey_id: surveyId, student_ids: [pending.student_id], use_template: templateName })
  });

  // 3. Update pending
  await supabase.from('pending_student_assignments').update({
    resolved_at: new Date().toISOString(),
    resolved_by: currentUser.email,
    resolved_cohort_id: cohortId
  }).eq('id', pendingId);

  // 4. Optional permanent mapping
  if (createMapping) {
    const acProductId = pending.ac_payload.product_id;
    await supabase.from('ac_product_mappings').insert({
      ac_product_id: acProductId, cohort_id: cohortId, survey_id: surveyId,
      template_name: templateName, active: true
    });
  }
}
```
